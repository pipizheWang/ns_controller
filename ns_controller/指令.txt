终端1：启动飞机1 UAV0
cd ~/PX4-Autopilot
conda activate torch
source /opt/ros/jazzy/setup.bash

export PX4_GZ_MODEL=x500
export PX4_SYS_AUTOSTART=4001
export PX4_GZ_MODEL_POSE="0,0,0.2,0,0,0"
./build/px4_sitl_default/bin/px4 -i 0
